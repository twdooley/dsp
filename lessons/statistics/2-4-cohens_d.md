    print('Means')
    print(f'First birth: {mean1}')
    print(f'Others: {mean_oth}\n')
    print('Variances')
    print(f'First: {var1}')
    print(f'Others: {var2}\n')
    print('Differences in Means')
    print(f'First - Others in lbs: {mean1 - mean_oth}')
    print(f'First - Others in oz: {(mean1-mean_oth)*16}\n')
